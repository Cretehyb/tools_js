	//快速排序
	function quickSort(arr){
	  	  if(arr.length==0){
	         return [];//判断数组长度为0时,直接返回空数组
	     }
	         var left=[]; //左边数组
	         var right=[];//右边数组
	         var base=arr[0]; //数组元素头部作为基准值
	
	      for(var i=1; i<arr.length; i++){
	        if(arr[i] < base){  //如果其他数组项小于基准值就分为左边，反之就分为右边
	          left.push(arr[i]);
	     }else{
	          right.push(arr[i]);
	     }
	 }
	 return quickSort(left).concat(base ,quickSort(right));//递归
 }
